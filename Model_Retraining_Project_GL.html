# DiscountOffer. First fill empty values with "0". Since there are "...% OFF", "... OFF" AND "0" in this 
# column, I change "... OFF" to "... % OFF", change "... % OFF" to float and change others to 0. 
# Then calculate the price after discount by OriginalPrice*(1-DiscountOffer)

df["DiscountOffer"].fillna("0",inplace=True)

def disRs_sep(discount):
    s = discount.split(" ")[1]
    return float(s)

def disper_sep(discount):
    s = discount.split("%")[0] 
    return float(s) 

d1 = df[df["DiscountOffer"].str.contains("%")]
d2 = df[df["DiscountOffer"].str.contains("Rs.")]
d3 = df[df["DiscountOffer"] == "0"]

d1["DiscountOffer"] = round(d1["DiscountOffer"].apply(disper_sep)/100,2)
d2["DiscountOffer"] = d2["DiscountOffer"].apply(disRs_sep)
d2["DiscountOffer"] = round(d2["DiscountOffer"] / d2["OriginalPrice"],2)
d3["DiscountOffer"] = round(0,2)
In [13]:
# concat them together
df1 = pd.concat([d1,d2,d3])
df1.sort_index(inplace = True)

df2 = pd.concat([d4,d5])
df2.sort_index(inplace = True)
